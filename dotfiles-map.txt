# Dotfiles symlink mapping
# Format: source_path=target_path
# Variables like $HOME and ~ are supported

# Shell configurations
~/.dotfiles/config/shells/bash/bashrc=~/.bashrc
~/.dotfiles/config/shells/bash/profile=~/.profile
~/.dotfiles/config/shells/zsh/zshrc=~/.zshrc

# Fish shell (special - entire directory goes in .config)
~/.dotfiles/config/shells/fish=~/.config/fish

# Editor configurations
~/.dotfiles/config/editors/nvim=~/.config/nvim

# Terminal configurations
~/.dotfiles/config/terminal/tmux=~/.config/tmux

# Tool configurations
~/.dotfiles/config/tools/git/gitconfig=~/.gitconfig

# Claude Code configurations
~/.dotfiles/config/tools/claude/settings.json=~/.claude/settings.json
~/.dotfiles/config/tools/claude/settings.local.json=~/.claude/settings.local.json
~/.dotfiles/config/tools/claude/CLAUDE.md=~/.claude/CLAUDE.md
~/.dotfiles/config/tools/claude/statusline-powerline.sh=~/.claude/statusline-powerline.sh
~/.dotfiles/config/tools/claude/plugins/config.json=~/.claude/plugins/config.json

~/.dotfiles/.hushlogin=~/.hushlogin
